112bool movementMgr_checkPosition(Direction direction, MovementGfx movGfx){
113
114 return (direction == DIRECTION_FORWARDS && (movementInfo[movGfx]->sprite->spriteEntry->x >= movementInfo[movGfx]->destinyPos.x
115 && movementInfo[movGfx]->sprite->spriteEntry->y >= movementInfo[movGfx]->destinyPos.y)) ||
116 (direction == DIRECTION_BACKWARDS && (movementInfo[movGfx]->sprite->spriteEntry->x <= movementInfo[movGfx]->destinyPos.x
117 && movementInfo[movGfx]->sprite->spriteEntry->y <= movementInfo[movGfx]->destinyPos.y));
118}

125bool movementMgr_nextPositionReached(MovementGfx movGfx){
126
127 if(movementMgr_checkPosition(movementInfo[movGfx]->direction, movGfx)){
128 movementInfo[movGfx]->posId += 1 * movementInfo[movGfx]->direction;
129 return true;
130 }else{
131 if(movGfx == MOVEMENT_INATRIX_X)
132 movementInfo[movGfx]->sprite->spriteEntry->x += 1 * movementInfo[movGfx]->direction;
133 else
134 movementInfo[movGfx]->sprite->spriteEntry->y += 1 * movementInfo[movGfx]->direction;
135 }
136
137 return false;
138}
