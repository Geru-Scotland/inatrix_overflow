41void timer_ConfigureTimer(int latch, int mask)
42{
43 timer_EnableInterruptions();
44
45 timer.ticks = 0;
46 timer.latch = latch;
47 timer.conf = mask;
48 timer.time = 0;
49 timer.totalTicks = 0;
50
51 TIMER0_CNT |= 0x00C0 | timer.conf;
52 TIMER0_DAT |= timer.latch;
53 timer_StartTimer();
54}

60void timer_UpdateTimer()
61{
62 timer.ticks++;
63 timer.totalTicks++;
64 eventMgr_UpdateScheduledEvents();
65 eventMgr_UpdateAnimations();
66 if(timer.ticks == TIMER0_FREQ){
67 timer.time++; // Seconds++
68 timer.ticks = 0;
69 }
70}
