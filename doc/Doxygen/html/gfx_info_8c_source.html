308void gfxInfo_initMatrix(Binary *base, uint8 size){
309 for(int i = 0; i < size; i++){
310 for(int j = 0; j < size; j++){
311 gfxInfo_allocateMatrixElement(*(base + i*size + j) ? GFX_DIGIT_ONE : GFX_DIGIT_ZERO);
312 gfxInfo_linkToMatrix(size, i, j, *(base + i*size + j) ? BIT_ONE : BIT_ZERO);
313 }
314 }
315}
