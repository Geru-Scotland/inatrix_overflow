60void eventMgr_DeleteEvent(Event *event){
61 uint8 i = event->pos;
62
63#ifdef DEBUG_MODE
64 iprintf("\x1b[%i;00H [DEL] NE: %i e.pos: %i - e.id: %i", lineDelete, numEvents, event->pos, event->id);
65 lineDelete += 1;
66#endif // DEBUG_MODE
67
68 while ((i < numEvents) && (eventList[i] != NULL))
69 {
70 eventList[i] = eventList[i+1];
71 eventList[i]->pos = i;
72 i++;
73 }
74
75 free(event);
76 numEvents--;
77#ifdef DEBUG_MODE
78 iprintf("\x1b[%i;00H [ARRAY] ={%i, %i, %i, %i}", lineDelete+6, eventList[0]->id, eventList[1]->id,eventList[2]->id,eventList[3]->id);
79 lineDelete += 1;
80#endif // DEBUG_MODE
81}

113void eventMgr_ScheduleEvent(uint8 eventId, int time){
114 /*
115 * Geru: Reservo memoria dinámica en el heap, porque el stack se vacía
116 * Una vez terminado el scope de la función, y queremos que persistan.
117 * Importante liberar la memoria reservada correspondiente a cada evento.
118 */
119 if(numEvents < MAX_EVENTS) {
120 Event* e = malloc(sizeof(Event));
121 e->id = eventId;
122 e->execTime = timer.time + time;
123 eventMgr_AddEvent(e);
124 }
125}

377void eventMgr_UpdatePhases(){
378 if((timer.ticks % 15 != 0) || gameData.state == GAME_STATE_PAUSE)
379 return;
380
381 switch(gameData.phase){
382 case PHASE_BITBLOCK_FALLING:
383 if(!matrix_dropBitBlockEffect()){
384 // Si vas a hacer el efecto de spawn desde diferentes posiciones
385 // Que se vean como 0,5sec después de que comience a caer el bitblock
386 eventMgr_ScheduleEvent(EVENT_GAME_REGENERATE_BITBLOCK, IN_1_SECONDS);
387 gameData.phase = PHASE_NULL;
388 }
389 break;
390 case PHASE_DESTROYING_MATRIX:
391 if(!matrix_destroyMatrixEffect()){
392 eventMgr_ScheduleEvent(EVENT_GAME_REGENERATE_MATRIX, IN_3_SECONDS);
393 gameData.phase = PHASE_NULL;
394 }
395 break;
396 case PHASE_MOVE_INATRIX_X:
397 if(movementMgr_nextPositionReached(MOVEMENT_INATRIX_X)){
398 matrix_updatePivot(movementMgr_getPositionY(), movementMgr_getPositionX());
399 objectMgr_setAnimationActive(ANIMATION_BIT_SHAKE, true);
400 gameData.phase = PHASE_WAITING_PLAYER_INPUT;
401 }
402 break;
403 case PHASE_MOVE_INATRIX_Y:
404 if(movementMgr_nextPositionReached(MOVEMENT_INATRIX_Y)){
405 matrix_updatePivot(movementMgr_getPositionY(), movementMgr_getPositionX());
406 objectMgr_setAnimationActive(ANIMATION_BIT_SHAKE, true);
407 gameData.phase = PHASE_WAITING_PLAYER_INPUT;
408 }
409 break;
410 case PHASE_MOVE_CAPSULE:
411 if(movementMgr_hasGfxReachedDest(
412 gameData.mode == DIFFICULTY_NORMAL_MODE ? GFX_CAPSULE_BLUE : GFX_CAPSULE_RED)){
413 gameData.phase = PHASE_NULL;
414 }
415 break;
416 default:
417 break;
418 }
419
420 oamUpdate(&oamMain);
421}
