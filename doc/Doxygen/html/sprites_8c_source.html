111void sprites_dumpToMemory(GfxData* gfx){
112 int gfxSize = (gfx->size == SpriteSize_16x16) ? (16 * 16)/2 : (32 * 32)/2;
113 for(int pos = 0; pos < gfxSize; pos++){
114 gfx->memAddress[pos] = gfx->bitmap[pos * 2] | (gfx->bitmap[(pos*2)+1] << 8);
115 }
116}

185void sprites_updateSprite(uint8 index){
186 oamSet(&oamMain, //main graphics engine context
187 index, //oam index (0 to 127)
188 sprites[index]->spriteEntry->x, sprites[index]->spriteEntry->y, //x and y pixel location of the sprite
189 0, //priority, lower renders last (on top)
190 0, //this is the palette index if multiple palettes or the alpha value if bmp sprite
191 sprites[index]->gfx->size,
192 sprites[index]->gfx->colorFormat,
193 sprites[index]->gfx->memAddress, //+16*16/2, //pointer to the loaded graphics
194 -1, //sprite rotation data
195 false, //double the size when rotating?
196 sprites[index]->spriteEntry->isHidden, //hide the sprite?
197 false, false, //vflip, hflip
198 false //apply mosaic
199 );
200 oamUpdate(&oamMain);
201}
