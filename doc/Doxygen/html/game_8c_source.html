79void game_Loop()
80{
81 game_initData();
82 game_launch();
83
84 while(SWITCH)
85 {
86 game_Update();
87
88 switch(gameData.state){
89 case GAME_STATE_MAIN_MENU:
90 switch(gameData.phase) {
91 case PHASE_WAITING_PLAYER_INPUT:
92 if (keyData.isPressed) {
93 gameData.phase = PHASE_NULL;
94 gameData.state = GAME_STATE_INTRO;
95 eventMgr_ScheduleEvent(EVENT_CLEAR_CONSOLE, NO_WAIT);
96 eventMgr_ScheduleEvent(EVENT_INTRO_PRE_START, IN_2_SECONDS);
97 }
98 break;
99 default:
100 break;
101 }
102 break;
103 case GAME_STATE_INTRO:
104 switch(gameData.phase){
105 case PHASE_INTRO_START:
106 eventMgr_ScheduleEvent(EVENT_MAIN_MENU_START, IN_2_SECONDS);
107 eventMgr_ScheduleEvent(EVENT_INTRO_SETBG_2, IN_2_SECONDS);
108 gameData.phase = PHASE_INTRO_SCENE_ACTIVE;
109 break;
110 case PHASE_WAITING_PLAYER_INPUT:
111 if(input_touchScreenUsed()){
112 int capsule = objectMgr_objectAreaClicked(input_getTouchScreenX(), input_getTouchScreenY());
113 if(capsule != -1){
114 if(capsule == GFX_CAPSULE_RED){
115 gameData.mode = DIFFICULTY_HARD_MODE;
116 }else{
117 gameData.mode = DIFFICULTY_NORMAL_MODE;
118 }
119 gameData.phase = PHASE_NULL;
120 eventMgr_ScheduleEvent(EVENT_INTRO_CAPSULE_SELECTED, NO_WAIT);
121 }
122 }
123 break;
124 default:
125 break;
126 }
127 break;
128 case GAME_STATE_GAME:
129 switch(gameData.phase){
130 case PHASE_WAITING_PLAYER_INPUT:
131 if(keyData.isPressed){
132 switch(keyData.key){
133 case INPUT_KEY_LEFT:
134 movementMgr_updateDirection(MOVEMENT_INATRIX_X, DIRECTION_BACKWARDS);
135 eventMgr_ScheduleEvent(EVENT_GAME_INATRIX_MOVE_X, NO_WAIT);
136 break;
137 case INPUT_KEY_RIGHT:
138 movementMgr_updateDirection(MOVEMENT_INATRIX_X, DIRECTION_FORWARDS);
139 eventMgr_ScheduleEvent(EVENT_GAME_INATRIX_MOVE_X, NO_WAIT);
140 break;
141 case INPUT_KEY_DOWN:
142 movementMgr_updateDirection(MOVEMENT_INATRIX_Y, DIRECTION_FORWARDS);
143 eventMgr_ScheduleEvent(EVENT_GAME_INATRIX_MOVE_Y, NO_WAIT);
144 break;
145 case INPUT_KEY_UP:
146 movementMgr_updateDirection(MOVEMENT_INATRIX_Y, DIRECTION_BACKWARDS);
147 eventMgr_ScheduleEvent(EVENT_GAME_INATRIX_MOVE_Y, NO_WAIT);
148 break;
149 case INPUT_KEY_A:
150 eventMgr_ScheduleEvent(EVENT_GAME_EVALUATE_BITBLOCK, NO_WAIT);
151 break;
152 case INPUT_KEY_START:
153 // Libnds mantiene por más de un tick datos de teclas
154 // Esto es para evitar los problemas que se generan por
155 // ello.
156 if(gameData.destroyMatrixTime > 2) // Temporal, para evitar problemas con la regen.
157 eventMgr_ScheduleEvent(EVENT_GAME_PAUSE, IN_1_SECONDS);
158 break;
159 default:
160 break;
161 }
162 gameData.phase = PHASE_NULL;
163 }
164 break;
165 default:
166 break;
167 }
168 break;
169 case GAME_STATE_PAUSE:
170 if(gameData.phase == PHASE_GAME_PAUSE){
171 if(keyData.isPressed && (keyData.key == INPUT_KEY_START)){
172 gameData.state = GAME_STATE_GAME;
173 eventMgr_ScheduleEvent(EVENT_LISTEN_INPUT, IN_1_SECONDS);
174 }
175 }
176 break;
177 case GAME_STATE_STATS:
178 if((gameData.phase == PHASE_SHOW_STATS) && keyData.isPressed){
179 game_initData();
180 game_launch();
181 SWITCH = 0;
182 }
183 break;
184 default:
185 break;
186 }
187 }
188
189 sprites_freeMemory();
190 gfxInfo_freeMemory();
191}

198void game_manageScore(bool overflow){
199 if(overflow){
200 playerData.overflowScore++;
201 playerData.totalOverflows++;
202 playerData.runOverflows++;
203 }else{
204 playerData.overflowScore--;
205 playerData.failScore++;
206 if(playerData.overflowScore < 0){
207 game_manageGameOver();
208 return;
209 }
210 }
211 consoleUI_showUI();
212}
